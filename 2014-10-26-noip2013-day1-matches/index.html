NOIP2013 Day1 зҒ«жҹҙжҺ’йҳҹпјҡеҝ«йҖҹжұӮйҖҶеәҸеҜ№

2014-10-26 | Tech

йўҳзӣ®

ж¶өж¶өжңүдёӨзӣ’зҒ«жҹҙпјҢжҜҸзӣ’иЈ…жңү n ж №зҒ«жҹҙпјҢжҜҸж №зҒ«жҹҙйғҪжңүдёҖдёӘй«ҳеәҰгҖӮзҺ°еңЁе°ҶжҜҸзӣ’дёӯзҡ„зҒ«жҹҙеҗ„иҮӘжҺ’жҲҗдёҖеҲ—пјҢеҗҢдёҖеҲ—зҒ«жҹҙзҡ„й«ҳеәҰдә’дёҚзӣёеҗҢпјҢдёӨеҲ—зҒ«жҹҙд№Ӣй—ҙзҡ„и·қзҰ»е®ҡд№үдёәпјҡ$\sum_{i=1}^{n}{(a_i-b_i)^2}$пјҢе…¶дёӯ aiиЎЁзӨәз¬¬дёҖеҲ—зҒ«жҹҙдёӯз¬¬ i дёӘзҒ«жҹҙзҡ„й«ҳеәҰпјҢbiиЎЁзӨәз¬¬дәҢеҲ—зҒ«жҹҙдёӯз¬¬ i дёӘзҒ«жҹҙзҡ„й«ҳеәҰгҖӮжҜҸеҲ—зҒ«жҹҙдёӯзӣёйӮ»дёӨж №зҒ«жҹҙзҡ„дҪҚзҪ®йғҪеҸҜд»ҘдәӨжҚўпјҢиҜ·дҪ йҖҡиҝҮдәӨжҚўдҪҝеҫ—дёӨеҲ—зҒ«жҹҙд№Ӣй—ҙзҡ„и·қзҰ»жңҖе°ҸгҖӮиҜ·й—®еҫ—еҲ°иҝҷдёӘжңҖе°Ҹзҡ„и·қзҰ»пјҢжңҖе°‘йңҖиҰҒдәӨжҚўеӨҡе°‘ж¬ЎпјҹеҰӮжһңиҝҷдёӘж•°еӯ—еӨӘеӨ§пјҢиҜ·иҫ“еҮәиҝҷдёӘжңҖе°ҸдәӨжҚўж¬Ўж•°еҜ№ 99,999,997 еҸ–жЁЎзҡ„з»“жһңгҖӮ

еҲҶжһҗ

иҝҷзңҹжҳҜдёҖйҒ“еҘҪйўҳвҖ”вҖ”ж–ӯж–ӯз»ӯз»ӯжғідәҶеҮ еӨ©жүҚе®Ңе…ЁACгҖӮдәӢе®һдёҠпјҢз”ұжҺ’еәҸдёҚзӯүејҸеҸҜзҹҘпјҡ

еҪ“$a_i, b_i$д»Һе°ҸеҲ°еӨ§жҺ’еәҸж—¶пјҢи·қзҰ»жңҖе°ҸгҖӮ

иҝҷжҳҜдёҖдёӘйҮҚиҰҒзҡ„дҝЎжҒҜгҖӮеӣ жӯӨпјҢжҲ‘д»¬еҸӘйңҖжҠҠ$a_i,b_i$иҝӣиЎҢжҺ’еәҸпјҢе№¶жҠҠеҜ№еә”йЎ№вҖңжҚҶз»‘вҖқжҲҗдёҖйЎ№пјҢеҶҚжҢү$a_i$еҺҹжңүзҡ„йЎәеәҸиҝӣиЎҢеӨҚеҺҹпјҢжӯӨж—¶пјҢеҸҜд»Ҙеҫ—еҲ°з”ұ%b_i$еҺҹе…Ҳзҡ„дёӢж Үз»„жҲҗзҡ„дёҖдёӘеәҸеҲ—гҖӮд№ҹе°ұжҳҜиҜҙпјҢжҲ‘д»¬иҰҒжұӮ$1,2,\ldots,n$иҮіе°‘з»ҸиҝҮеӨҡе°‘жӯҘжүҚиғҪеҸҳдёәиҜҘеәҸеҲ—гҖӮиҝҷеҸҜд»Ҙз”ЁйҖҶеәҸеҜ№жқҘи§ЈеҶігҖӮ еҸӘеҸҜжғңпјҢдј з»ҹзҡ„йҖҶеәҸеҜ№з®—жі•ж—¶й—ҙеӨҚжқӮеәҰдёә$O(n^2)$пјҢиҝҷйҮҢnеҸҜиҫҫ20,0000пјҢдёҖе®ҡдјҡи¶…ж—¶пјҲдәӢе®һдёҠпјҢеҸӘиҝҮдәҶ70%зҡ„зӮ№пјүгҖӮеӣ жӯӨжҲ‘д»¬йңҖеҜ»жұӮжӣҙеҘҪзҡ„з®—жі•гҖӮ

з”ЁеҪ’е№¶жҺ’еәҸжұӮйҖҶеәҸеҜ№

еңЁеҪ’е№¶жҺ’еәҸзҡ„иҝҮзЁӢдёӯпјҢжңүдёҖдёӘжӯҘйӘӨз§°дёәеҗҲе№¶гҖӮеңЁиҝҷдёӘжӯҘйӘӨдёӯпјҢйңҖиҰҒиҪ®жөҒеҲӨж–ӯе·ҰеҸіеҢәй—ҙзҡ„з¬¬дёҖдёӘж•°зҡ„еӨ§е°Ҹе…ізі»гҖӮжіЁж„ҸеҲ°пјҡе·ҰеҸіеҢәй—ҙе·Із»ҸжңүеәҸпјҢд»ҺиҖҢиӢҘе·ҰеҢәй—ҙзҡ„з¬¬дёҖдёӘж•°еӨ§дәҺеҸіеҢәй—ҙзҡ„з¬¬дёҖдёӘж•°пјҢеҲҷе·ҰеҢәй—ҙд№ӢеҗҺзҡ„жүҖжңүж•°йғҪеӨ§дәҺеҸіеҢәй—ҙзҡ„з¬¬дёҖдёӘж•°пјҢд»ҺиҖҢжҲ‘д»¬еҸҜд»ҘеңЁеҗҲе№¶ж—¶еҒҡдёҖдәӣдҝ®ж”№пјҡ

procedure nx(l, r: longint);
var
    mid, i, j, k: longint;
begin
    if l = r then
    begin
        tmp[l] := a[l];
        exit;
    end;
    mid := (l + r) shr 1;
    nx(l, mid);
    nx(mid+1, r);
    i := l;
    j := mid+1;
    k := l;
    while k <= r do
    begin
        if (j>r) or (i<=mid) and (a[i]<=a[j]) then //жіЁж„ҸиҝҷйҮҢдёәзӯүеҸ·
        begin
            tmp[k] := a[i];
            inc(i);
        end
        else
        begin
            cnt := cnt + mid - i + 1; //еҠ дёҠйҖҶеәҸж•°
            tmp[k] := a[j];
            inc(j);
        end;
        inc(k);
    end;
    for i := l to r do
        a[i] := tmp[i];
end;

иҝҷдёӘз®—жі•еӨҚжқӮеәҰдёә$O(nlogn)$пјҢжҳҜдёҖз§ҚжҜ”иҫғзҗҶжғізҡ„з®—жі•пјҢе®һзҺ°иө·жқҘд№ҹз®ҖеҚ•гҖӮдҪҶд»–жңүдёӘзјәзӮ№пјҡдјҡжү“д№ұеҺҹж•°з»„йЎәеәҸгҖӮ

//AC
const
    MODN = 99999997;
type
    rec = record value, index: longint; end;
    TArr = array [1..100000] of rec;
var
    n: longint;
    a, b, c, tmp: TArr;
    ok: Boolean;
    l, r, i, j: longint;
    cnt: int64;

procedure sort(var arr: TArr; l, r: longint);
var
    i, j: longint;
    m, t: rec;
begin
    i := l;
    j := r;
    m := arr[(i+j) shr 1];
    repeat
        while arr[i].value < m.value do inc(i);
        while arr[j].value > m.value do dec(j);
        if i <= j then
        begin
            t := arr[i];
            arr[i] := arr[j];
            arr[j] := t;
            inc(i);
            dec(j);
        end;
    until i >j;
    if i < r then sort(arr, i, r);
    if l < j then sort(arr, l, j);
end;

procedure nx(l, r: longint);
var
    mid, i, j, k: longint;
begin
    if l = r then
    begin
        tmp[l] := c[l];
        exit;
    end;
    mid := (l + r) shr 1;
    nx(l, mid);
    nx(mid+1, r);
    i := l;
    j := mid+1;
    k := l;
    while k <= r do
    begin
        if (j>r) or (i<=mid) and (c[i].index<=c[j].index) then
        begin
            tmp[k] := c[i];
            inc(i);
        end
        else
        begin
            cnt := cnt + mid - i + 1;
            cnt := cnt mod MODN;
            tmp[k] := c[j];
            inc(j);
        end;
        inc(k);
    end;
    for i := l to r do
        c[i] := tmp[i];
end;

begin
    assign(input, 'main.in'); reset(input);
    assign(output, 'main.out'); rewrite(output);

    readln(n);
    for i := 1 to n do
    begin
        read(a[i].value);
        a[i].index := i;
    end;
    for i := 1 to n do
    begin
        read(b[i].value);
        b[i].index := i;
    end;
    sort(a, 1, n);
    sort(b, 1, n);

    for i := 1 to n do
        c[a[i].index] := b[i];

    cnt := 0;
    nx(1, n);
    writeln(cnt);

    close(input); close(output);
end.
